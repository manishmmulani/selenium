IRCTC_passengers
storeEval var nickStr=prompt("Enter passengers comma-separated " + getNicks()); nickStr.split(",") passengerNicks
storeEval storedVars['passengerNicks'].length noOfPassengers
storeEval confirm("Tatkal?") isTatkal
store 0 passengerIndex
while ${passengerIndex} < ${noOfPassengers}
storeEval getPassenger(storedVars['passengerNicks'][storedVars['passengerIndex']]) currentPassenger
type id=addPassengerForm:psdetail:${passengerIndex}:psgnName javascript{storedVars['currentPassenger'].name}
type id=addPassengerForm:psdetail:${passengerIndex}:psgnAge javascript{storedVars['currentPassenger'].age}
select id=addPassengerForm:psdetail:${passengerIndex}:psgnGender javascript{storedVars['currentPassenger'].gender}
gotoIf !${isTatkal} seniority
select id=addPassengerForm:psdetail:${passengerIndex}:idCardType PAN Card
type id=addPassengerForm:psdetail:${passengerIndex}:idCardNumber javascript{storedVars['currentPassenger'].pan}
label seniority
gotoIf javascript{storedVars['isTatkal'] || !storedVars['currentPassenger'].senior} loopOver
click id=addPassengerForm:psdetail:${passengerIndex}:concessionOpt
label loopOver
storeEval ${passengerIndex} + 1 passengerIndex
endWhile
click id=addPassengerForm:autoUpgrade
type id=addPassengerForm:mobileNo 9160635930
storeEval prompt("Captcha?") captcha
type id=j_captcha ${captcha}
storeEval confirm("Can I proceed with the booking?") confirmBooking
gotoIf !${confirmBooking} invalidForm
clickAndWait id=validate
label invalidForm
